#include <iostream>

using namespace std;

int main()
{
    //Arithmetic Operators
    int addition = 10 + 10; // +
    cout << addition << endl;
    int subtraction = 10 - 10; // -
    cout << subtraction << endl;
    int division = 25 / 5; // /
    cout << division << endl;
    int multiplication = 10 * 10; // *
    cout << multiplication << endl;
    int modulo = 17 % 35; // %
    cout << modulo << endl;
    

    return 0;
}
